//2119 최시율 제작--코드는 작년에 짠 코드입니다.
#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <cstring>
#pragma warning(disable:4996)
using namespace std;
 
int n, m;
//맨 처음 칸을 0으로 두고 순서대로 번호를 붙임
/*
0 1 2 3
4 5 6 7 이런 식으로...
*/
int dp[200][1 << 15];//여기에다가 경우의 수를 저장
 
int go(int N, int S)//N번째 칸부터 m개 칸의 상태가 S(비트마스크)일 때 경우의 수
{
    if (N == n * m&& S == 0)//마지막 칸에서 S가 0=>아래로 안 삐져나오고 제대로 채웠다
        return 1;
 
    if (N >= n * m)
        return 0;
 
    int &fin = dp[N][S];//참조형 변수->시간 단축용!
    if (fin != -1)
        return fin;
 
    fin = 0;
 
    if ((S & 1) == 1)//이미 N번째 칸이 채워져 있는 경우
    {
        fin = go(N + 1, S >> 1);
        fin %= 9901;
    }
    else
    {
        //2*1(세로)을 넣는 경우
        //N+m칸 비트를 켠 뒤 N+1번째 칸으로 이동
        fin = go(N + 1, (S | (1 << m)) >> 1);
        fin %= 9901;
 
        //1*2(가로)를 넣는 경우->다음 칸이 비어 있는지 검사
        //N번째 칸이 가장 오른쪽 칸이면 1*2를 넣을 수 없으므로 그것도 검사
        if ((S & 2) == 0 && (N%m) + 1 != m)
        {
            fin += go(N + 2, S >> 2);//모든 경우의 수를 구하는 것이므로 max가 아니라 더함
            fin %= 9901;
        }
    }
 
    return fin % 9901;
 
 
 
}
 
int main()
{
    memset(dp, -1, sizeof(dp));
    scanf("%d %d", &n, &m);
    printf("%d", go(0, 0));
}
